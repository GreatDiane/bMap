              if(ret.status){

            });

setRegion 设置百度地图

            var map = api.require('bMap');
            map.setRegion({
                lbLon: 116.39432327,
                lbLat: 39.88933191,
                rtLon: 116.49432328,
                rtLat: 39.98963192,
                animation: true
            });

setRotation 设置百度地图

            var map = api.require('bMap');
            map.setRotation({
              degree: 100
            });

setOverlook 设置百度地图

            var map = api.require('bMap');
            map.setOverlook({
               degree: -30
            });

setAttr 设置百度地图

            var map = api.require('bMap');
            map.setMapAttr({
               type: 'satellite',
               zoomEnable:false,
               scrollEnable:false
            });

setZoomLevel 设置百度地图缩放等级

            var map = api.require('bMap');
            map.setZoomLevel({
               level:13 // 地图比例尺级别,当前手机可用 3 - 18级.
            });

setCenter 设置百度地图中心点

            var map = api.require('bMap');
            map.setCenter({
                coords:{
                    lon:116.38429, // 经度
                    lat:39.98958 // 纬度
                }
            });

getDistance 获取两点之间的距离

            var map = api.require('bMap');
            map.getDistance({
                start: {
                    lon: 106.486654,
                    lat: 29.490295
                },
                end: {
                    lon: 106.581515,
                    lat: 29.615467
                }
            },function(ret){
                if(ret.status){
                    api.alert({msg:ret.distance});
                }
            });

showUserLocation 设置是否在地图上显示用户位置

            var map = api.require('bMap');
            map.showUserLocation({
                isShow:true, // true,显示用户位置;false,不显示用户位置.
                trackingMode:'compass' // 跟踪类型,可选: none,标准模式; follow, 跟踪模式; compass, 罗盘模式.
            });

hide 设置是否隐藏地图

            var map = api.require('bMap');
            
            map.hide();

show 设置是否隐藏地图

            var map = api.require('bMap');
            
            map.show();

close 关闭地图

            var map = api.require('bMap');
            
            map.close();
